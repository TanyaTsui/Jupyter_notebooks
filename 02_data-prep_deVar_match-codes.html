  • some addresses are duplicated, because multiple companies might be located in the same address.

def match_by_geo_text(remaining, kvk, buffer, fuzzRatio): 
    # geo-locate kvk dataset using pc6_points
    kvkLoc = pd.merge(kvk, pc6_points[['postcode6', 'geometry']], how='left', left_on='pc', right_on='postcode6')
    kvkLoc = kvkLoc[['pc', 'postcode6', 'naam', 'address', 'sbi', 'geometry']]
    kvkLoc = kvkLoc[~kvkLoc.geometry.isna()]
    kvkLoc = gpd.GeoDataFrame(kvkLoc, geometry='geometry')
    kvkLoc = kvkLoc.to_crs('epsg:28992') # reproject to amersfoort
    
    # geo-locate remaining lma dataset 
    remaining = remaining[['eaNaam', 'eaPostcode', 'kg']]
    remaining = pd.merge(remaining, pc6_points[['postcode6', 'geometry']], how='left', left_on='eaPostcode', right_on='postcode6')
    remaining = remaining[~remaining.geometry.isna()]
    remaining = gpd.GeoDataFrame(remaining)
    remaining = remaining.to_crs('EPSG:28992')

    # create bufferzones of 500m 
    remaining['buffer500m'] = remaining.geometry.buffer(buffer)
    buffer = gpd.GeoDataFrame(remaining[['eaNaam', 'buffer500m']], geometry='buffer500m', crs={"init": "epsg:28992"})
    
    # spatial join of lma buffers and kvk company locations 
    compare = gpd.sjoin(buffer, kvkLoc, how='inner', op='intersects')
    compare.drop_duplicates(inplace=True)
    
    # calculate fuzz ratio between eerste afnemer and the kvk actors with 500m 
    def fuzz_ratio(row): 
        ratio = fuzz.ratio(row.eaNaam, row.naam)
        return ratio
    compare['fuzzRatio'] = compare.apply(lambda row: fuzz_ratio(row), axis=1)
    compare.sort_values('fuzzRatio', ascending=False).head()
    
    # select matches between lma and kvk dataset 
    matched = compare[compare.fuzzRatio >= fuzzRatio].reset_index() # select matches if they score higher than 60
    matched = matched.loc[matched.groupby(['eaNaam'])['fuzzRatio'].idxmax()] # select highest scoring match
    matched = matched.groupby('eaNaam').sbi.aggregate(lambda x: list(x)).reset_index()
    matched_by_fuzzy = pd.merge(remaining_by_address, matched, how='left', on='eaNaam')
    
    # create outputs  
    output_by_fuzzy = matched_by_fuzzy[(matched_by_fuzzy.sbi.isna() == False) & (matched_by_fuzzy.sbi.str.len() == 1)]
    output_by_fuzzy['how'] = 'by proximity and fuzzy name'
    multimatch_by_fuzzy = matched_by_fuzzy[(matched_by_fuzzy.sbi.str.len() > 1)]
    remaining_by_fuzzy = matched_by_fuzzy[(matched_by_fuzzy.sbi.isna() == True)]
    remaining_by_fuzzy = remaining_by_fuzzy.drop(labels=['sbi'], axis=1)
    
    return output_by_fuzzy, multimatch_by_fuzzy , remaining_by_fuzzy, compare[compare.fuzzRatio >= fuzzRatio].reset_index()

def match_by_keyword(multimatch_all): 
    # PUBLIC ADMINISTRATION 
    # SBI code of '84' - Public administration, public services and compulsory social security
    govKeywords = ['GEMEENTE', 'POLITIE', 'RIJKSWATERSTAAT', 'HOGESCHOOL', 'STADSDEEL',
                'BELASTINGDIENST', 'PROVINCIE', 'OPENBARE', 'UNIE', 'UNIVERSITEIT',
                'RAADHUIS', 'WATERSCHAP']

    # separate output_all into gov and non-gov 
    multimatch_all.eaNaam = multimatch_all.eaNaam.str.upper()
    gov = multimatch_all[multimatch_all['eaNaam'].str.contains('|'.join(govKeywords))]
    nonGov = multimatch_all[~multimatch_all['eaNaam'].str.contains('|'.join(govKeywords))]
    
    # replace sbi code with '84' if there was no match / multiple sbis 
    gov.sbi.fillna('--', inplace=True)
    gov.sbi = gov.sbi.map(lambda x: ['84'] if len(x) > 1 or x == np.NaN else x)
    gov['how'] = 'by keyword'
    
    # create outputs
    output_by_keyword = gov
    multimatch_by_keyword = nonGov[nonGov.sbi.str.len() > 1]
    remaining_by_keyword = nonGov[nonGov.sbi.isna()]

    return output_by_keyword, multimatch_by_keyword, remaining_by_keyword

def match_by_sbichap(multimatch_by_keyword): 
    # make dictionary matching with sections as keys and chapters as items 
    sbiDesc = pd.read_feather('classification/sbi_Headings.feather')
    sbiDict = dict(sbiDesc.groupby(['section']).sbi.aggregate(lambda x: list(x)))
    for k, v in sbiDict.items(): 
        v = v[1:]
        sbiDict[k] = v
        
    by_sbichap = multimatch_by_keyword.copy()

    # convert mulsbi.sbi in section codes 
    def sbi(x): 
        # x = ['23871', '238792', ...]
        newx = []
        for sbi in x: 
            chap = sbi[:2]
            for sec, chaps in sbiDict.items(): 
                if chap in chaps: 
                    newx.append(sec)
        newx = list(set(newx))
        return newx # newx = list of section codes = ['A', 'A', 'C', ...]

    multimatch_by_sbichap = by_sbichap[by_sbichap.sbi.str.len() > 1]

Fixing ewc and gnc codes¶

How are the two material codes structured?

  • EWC codes can be 2, 4, or 6 digits, like 01, 0103, or 010302. '01' is the section code', '03' is the chapter code, and '02' is the item code.
  • GNC codes can be 2, 4, 6, or 8 digits, like 25, 2501, 250100, or 25010010. '25' is the chapter code, '01' is the sub-chapter code, '00' is the sub-sub-chapter code, and '10' is the item code. The chapters fall under sections, which are represented by roman numerals like 'XXI', 'V', or 'VI'

Some ewc codes in the dataset have 5 digits, which isn't a valid code length. They could be the cases where people wrote things like '7' instead of '07' for chapter 7.
